def bubble_sort(arr):
    n = len(arr)
    for i in range(n - 1):
        # 最后i个元素已经就位
        for j in range(0, n - i - 1):
            # 从0到n-i-1遍历
            if arr[j] > arr[j + 1]:
                # 交换元素
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
    return arr
def selection_sort(arr):
    n = len(arr)
    # 一个一个地将未排序部分的最小元素放到已排序部分的末尾
    for i in range(n - 1):
        # 找到未排序部分中的最小元素
        min_idx = i
        for j in range(i + 1, n):
            if arr[j] < arr[min_idx]:
                min_idx = j
        
        # 将找到的最小元素与未排序部分的第一个元素交换
        arr[i], arr[min_idx] = arr[min_idx], arr[i]
    return arr
def insertion_sort(arr):
    n = len(arr)
    for i in range(1, n):
        key = arr[i]
        j = i - 1
        
        # 将arr[0..i-1]中大于key的元素向后移动一位
        while j >= 0 and arr[j] > key:
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key
    return arr
def quick_sort(arr, low=0, high=None):
    if high is None:
        high = len(arr) - 1
    
    if low < high:
        # pi是分区点，arr[pi]现在位于正确的位置
        pi = partition(arr, low, high)
        
        # 分别对分区点左右两部分进行排序
        quick_sort(arr, low, pi - 1)
        quick_sort(arr, pi + 1, high)
    
    return arr

def partition(arr, low, high):
    # 选择最右边的元素作为枢轴
    pivot = arr[high]
    i = low - 1  # 小于枢轴的元素的索引
    
    for j in range(low, high):
        # 如果当前元素小于或等于枢轴
        if arr[j] <= pivot:
            i += 1
            # 交换arr[i]和arr[j]
            arr[i], arr[j] = arr[j], arr[i]
    
    # 交换arr[i+1]和arr[high]（或枢轴）
    arr[i + 1], arr[high] = arr[high], arr[i + 1]
    
    return i + 1
def merge_sort(arr, l=0, r=None):
    if r is None:
        r = len(arr) - 1
    
    if l < r:
        # 找到中间点
        m = l + (r - l) // 2
        
        # 对两半分别排序
        merge_sort(arr, l, m)
        merge_sort(arr, m + 1, r)
        
        # 合并已排序的两半
        merge(arr, l, m, r)
    
    return arr

def merge(arr, l, m, r):
    n1 = m - l + 1
    n2 = r - m
    
    # 创建临时数组
    L = [0] * n1
    R = [0] * n2
    
    # 复制数据到临时数组L[]和R[]
    for i in range(n1):
        L[i] = arr[l + i]
    for j in range(n2):
        R[j] = arr[m + 1 + j]
    
    # 合并临时数组
    i = 0  # 初始索引L[]
    j = 0  # 初始索引R[]
    k = l  # 初始索引合并后的子数组
    
    while i < n1 and j < n2:
        if L[i] <= R[j]:
            arr[k] = L[i]
            i += 1
        else:
            arr[k] = R[j]
            j += 1
        k += 1
    
    # 复制L[]的剩余元素
    while i < n1:
        arr[k] = L[i]
        i += 1
        k += 1
    
    # 复制R[]的剩余元素
    while j < n2:
        arr[k] = R[j]
        j += 1
        k += 1
def heap_sort(arr):
    n = len(arr)
    
    # 构建最大堆
    for i in range(n // 2 - 1, -1, -1):
        heapify(arr, n, i)
    
    # 一个个从堆顶取出元素
    for i in range(n - 1, 0, -1):
        # 将当前堆顶（最大值）移到数组末尾
        arr[i], arr[0] = arr[0], arr[i]
        
        # 对剩余的堆进行堆化
        heapify(arr, i, 0)
    
    return arr

# 堆化以root为根的子树，n是堆的大小
def heapify(arr, n, root):
    largest = root  # 初始化最大值为根
    left = 2 * root + 1  # 左子节点 = 2*root + 1
    right = 2 * root + 2  # 右子节点 = 2*root + 2
    
    # 如果左子节点大于根
    if left < n and arr[left] > arr[largest]:
        largest = left
    
    # 如果右子节点大于到目前为止的最大值
    if right < n and arr[right] > arr[largest]:
        largest = right
    
    # 如果最大值不是根
    if largest != root:
        arr[root], arr[largest] = arr[largest], arr[root]
        
        # 递归地堆化受影响的子树
        heapify(arr, n, largest)

复杂度分析

时间复杂度

时间复杂度表示算法执行所需的时间与输入规模的关系。

  • O(n) - 线性时间复杂度，如冒泡排序在最好情况下
  • O(n log n) - 线性对数时间复杂度，如快速排序、归并排序、堆排序
  • O(n²) - 平方时间复杂度，如冒泡排序、选择排序、插入排序

空间复杂度

空间复杂度表示算法执行所需的额外空间与输入规模的关系。

  • O(1) - 常数空间复杂度，如冒泡排序、选择排序、插入排序、堆排序
  • O(log n) - 对数空间复杂度，如快速排序
  • O(n) - 线性空间复杂度，如归并排序
